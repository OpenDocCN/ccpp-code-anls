/* yy_get_next_buffer - try to read in a new buffer
 *
 * Returns a code representing an action:
 *    EOB_ACT_LAST_MATCH - 表示已经匹配到最后一个字符
 *    EOB_ACT_CONTINUE_SCAN - 继续从当前位置扫描
 *    EOB_ACT_END_OF_FILE - 文件结束
 */
static int yy_get_next_buffer (yyscan_t yyscanner)
{
    struct yyguts_t * yyg = (struct yyguts_t*)yyscanner;  // 定义 yyguts_t 结构体指针 yyg，指向 yyscanner
    char *dest = YY_CURRENT_BUFFER_LVALUE->yy_ch_buf;  // 定义指向当前缓冲区的目标指针 dest
    char *source = yyg->yytext_ptr;  // 定义指向 yytext_ptr 的源指针 source
    yy_size_t number_to_move, i;  // 定义需要移动的字符数和循环变量 i
    int ret_val;  // 定义返回值

    if ( yyg->yy_c_buf_p > &YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[yyg->yy_n_chars + 1] )  // 如果当前缓冲区指针超出了缓冲区范围
        YY_FATAL_ERROR(  // 报错
        "fatal flex scanner internal error--end of buffer missed" );

    if ( YY_CURRENT_BUFFER_LVALUE->yy_fill_buffer == 0 )  // 如果不需要填充缓冲区，即已经到达文件末尾
        { /* Don't try to fill the buffer, so this is an EOF. */
        if ( yyg->yy_c_buf_p - yyg->yytext_ptr - YY_MORE_ADJ == 1 )  // 如果当前指针和文本指针之间的距离为 1
            {
            /* We matched a single character, the EOB, so
             * treat this as a final EOF.
             */
            return EOB_ACT_END_OF_FILE;  // 返回文件结束
            }

        else
            {
            /* We matched some text prior to the EOB, first
             * process it.
             */
            return EOB_ACT_LAST_MATCH;  // 返回匹配到最后一个字符
            }
        }

    /* Try to read more data. */

    /* First move last chars to start of buffer. */
    number_to_move = (yy_size_t) (yyg->yy_c_buf_p - yyg->yytext_ptr) - 1;  // 计算需要移动的字符数

    for ( i = 0; i < number_to_move; ++i )  // 循环移动字符
        *(dest++) = *(source++);

    if ( YY_CURRENT_BUFFER_LVALUE->yy_buffer_status == YY_BUFFER_EOF_PENDING )  // 如果缓冲区状态为待处理的 EOF
        /* don't do the read, it's not guaranteed to return an EOF,
         * just force an EOF
         */
        YY_CURRENT_BUFFER_LVALUE->yy_n_chars = yyg->yy_n_chars = 0;  // 强制将字符数置为 0
    else
        {

            yy_size_t num_to_read = YY_CURRENT_BUFFER_LVALUE->yy_buf_size - number_to_move - 1;

            // 当需要读取的字符数小于等于0时，扩大缓冲区
            while ( num_to_read <= 0 )
            { /* Not enough room in the buffer - grow it. */

                /* 获取当前缓冲区的引用 */
                YY_BUFFER_STATE b = YY_CURRENT_BUFFER_LVALUE;

                // 计算当前字符指针相对于缓冲区起始位置的偏移量
                int yy_c_buf_p_offset = (int) (yyg->yy_c_buf_p - b->yy_ch_buf);

                // 如果当前缓冲区是我们自己创建的
                if ( b->yy_is_our_buffer )
                {
                    // 计算新的缓冲区大小
                    yy_size_t new_size = b->yy_buf_size * 2;

                    // 如果新的大小小于等于0，增加缓冲区大小的1/8
                    if ( new_size <= 0 )
                        b->yy_buf_size += b->yy_buf_size / 8;
                    else
                        b->yy_buf_size *= 2;

                    // 重新分配缓冲区内存
                    b->yy_ch_buf = (char *) pcap_realloc((void *) b->yy_ch_buf, b->yy_buf_size + 2, yyscanner);
                }
                else
                    // 无法扩大缓冲区，因为不是我们自己创建的
                    b->yy_ch_buf = 0;

                // 如果缓冲区为空，抛出致命错误
                if ( ! b->yy_ch_buf )
                    YY_FATAL_ERROR("fatal error - scanner input buffer overflow");

                // 更新当前字符指针的位置
                yyg->yy_c_buf_p = &b->yy_ch_buf[yy_c_buf_p_offset];

                // 重新计算需要读取的字符数
                num_to_read = YY_CURRENT_BUFFER_LVALUE->yy_buf_size - number_to_move - 1;
            }

            // 如果需要读取的字符数大于指定的读取缓冲区大小，限制为指定的读取缓冲区大小
            if ( num_to_read > YY_READ_BUF_SIZE )
                num_to_read = YY_READ_BUF_SIZE;

            // 读取更多的数据
            YY_INPUT( (&YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[number_to_move]), yyg->yy_n_chars, num_to_read );

            // 更新当前缓冲区的字符数
            YY_CURRENT_BUFFER_LVALUE->yy_n_chars = yyg->yy_n_chars;
        }
    # 如果当前缓冲区的字符数为0
    if ( yyg->yy_n_chars == 0 )
        {
        # 如果需要移动的字符数为YY_MORE_ADJ
        if ( number_to_move == YY_MORE_ADJ )
            {
            # 设置返回值为EOB_ACT_END_OF_FILE
            ret_val = EOB_ACT_END_OF_FILE;
            # 重新初始化输入流
            pcap_restart(yyin  ,yyscanner);
            }

        else
            {
            # 设置返回值为EOB_ACT_LAST_MATCH
            ret_val = EOB_ACT_LAST_MATCH;
            # 设置当前缓冲区状态为YY_BUFFER_EOF_PENDING
            YY_CURRENT_BUFFER_LVALUE->yy_buffer_status =
                YY_BUFFER_EOF_PENDING;
            }
        }

    else
        # 设置返回值为EOB_ACT_CONTINUE_SCAN
        ret_val = EOB_ACT_CONTINUE_SCAN;

    # 如果当前字符数加上需要移动的字符数大于当前缓冲区大小

        # 扩展数组大小为当前字符数加上需要移动的字符数的50%，再加上实际需要的字符数

        # 重新分配缓冲区内存
        YY_CURRENT_BUFFER_LVALUE->yy_ch_buf = (char *) pcap_realloc((void *) YY_CURRENT_BUFFER_LVALUE->yy_ch_buf,new_size ,yyscanner );
        # 如果内存分配失败，则抛出错误
        if ( ! YY_CURRENT_BUFFER_LVALUE->yy_ch_buf )
            YY_FATAL_ERROR( "out of dynamic memory in yy_get_next_buffer()" );
    }

    # 更新字符数
    yyg->yy_n_chars += number_to_move;
    # 在缓冲区末尾添加结束符
    YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[yyg->yy_n_chars] = YY_END_OF_BUFFER_CHAR;
    YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[yyg->yy_n_chars + 1] = YY_END_OF_BUFFER_CHAR;

    yyg->yytext_ptr = &YY_CURRENT_BUFFER_LVALUE->yy_ch_buf[0];

    return ret_val;

/** Allocate and initialize an input buffer state.

 * @param yyscanner The scanner object.
 * @return the allocated buffer state.
 */
    YY_BUFFER_STATE pcap__create_buffer  (FILE * file, int  size , yyscan_t yyscanner)
{
    YY_BUFFER_STATE b;
    
    // 分配一个 yy_buffer_state 结构体的内存空间
    b = (YY_BUFFER_STATE) pcap_alloc(sizeof( struct yy_buffer_state ) ,yyscanner );
    // 如果分配失败，则抛出错误
    if ( ! b )
        YY_FATAL_ERROR( "out of dynamic memory in pcap__create_buffer()" );

    // 设置缓冲区的大小
    b->yy_buf_size = (yy_size_t)size;

    /* yy_ch_buf has to be 2 characters longer than the size given because
     * we need to put in 2 end-of-buffer characters.
     */
    // 分配缓冲区的内存空间，并且比给定的大小多两个字符
    b->yy_ch_buf = (char *) pcap_alloc(b->yy_buf_size + 2 ,yyscanner );

    b->yy_is_our_buffer = 1;

    pcap__init_buffer(b,file ,yyscanner);

    return b;

    // 将传入的扫描器对象转换为 yyguts_t 结构体指针
    struct yyguts_t * yyg = (struct yyguts_t*)yyscanner;

    // 如果缓冲区为空，则直接返回
    if ( ! b )
        return;

    // 如果缓冲区是当前缓冲区，则将当前缓冲区置为 0
    if ( b == YY_CURRENT_BUFFER ) /* Not sure if we should pop here. */
        YY_CURRENT_BUFFER_LVALUE = (YY_BUFFER_STATE) 0;

    // 如果缓冲区是程序分配的，则释放缓冲区的内存空间
    if ( b->yy_is_our_buffer )
        pcap_free((void *) b->yy_ch_buf ,yyscanner );

    // 释放缓冲区结构体的内存空间
    pcap_free((void *) b ,yyscanner );
}
/* Initializes or reinitializes a buffer.
 * This function is sometimes called more than once on the same buffer,
 * such as during a pcap_restart() or at EOF.
 */
static void pcap__init_buffer  (YY_BUFFER_STATE  b, FILE * file , yyscan_t yyscanner)
{
    // 保存当前的错误号
    int oerrno = errno;
    // 将缓冲区中的内容刷新
    pcap__flush_buffer(b ,yyscanner);

    // 设置缓冲区的输入文件
    b->yy_input_file = file;
    // 设置填充缓冲区的标志为1
    b->yy_fill_buffer = 1;

    /* If b is the current buffer, then pcap__init_buffer was _probably_
     * called from pcap_restart() or through yy_get_next_buffer.
     * In that case, we don't want to reset the lineno or column.
     */
    // 如果 b 是当前缓冲区，则不重置行号和列号
    if (b != YY_CURRENT_BUFFER){
        b->yy_bs_lineno = 1;
        b->yy_bs_column = 0;
    }

    // 设置交互式标志为0
    b->yy_is_interactive = 0;
    
    // 恢复之前保存的错误号
    errno = oerrno;
}

/** Discard all buffered characters. On the next scan, YY_INPUT will be called.
 * @param b the buffer state to be flushed, usually @c YY_CURRENT_BUFFER.
 * @param yyscanner The scanner object.
 */
void pcap__flush_buffer (YY_BUFFER_STATE  b , yyscan_t yyscanner)
{
    // 获取扫描器对象
    struct yyguts_t * yyg = (struct yyguts_t*)yyscanner;
    // 如果缓冲区为空，则直接返回
    if ( ! b )
        return;

    // 将缓冲区中的字符数设置为0
    b->yy_n_chars = 0;

    /* We always need two end-of-buffer characters.  The first causes
     * a transition to the end-of-buffer state.  The second causes
     * a jam in that state.
     */
    // 设置两个结束符到缓冲区中
    b->yy_ch_buf[0] = YY_END_OF_BUFFER_CHAR;
    b->yy_ch_buf[1] = YY_END_OF_BUFFER_CHAR;

    // 设置缓冲区的位置指针
    b->yy_buf_pos = &b->yy_ch_buf[0];

    // 设置在行首的标志为1
    b->yy_at_bol = 1;
    // 设置缓冲区状态为新的
    b->yy_buffer_status = YY_BUFFER_NEW;

    // 如果缓冲区是当前缓冲区，则加载缓冲区状态
    if ( b == YY_CURRENT_BUFFER )
        pcap__load_buffer_state(yyscanner );
}

/** Pushes the new state onto the stack. The new state becomes
 *  the current state. This function will allocate the stack
 *  if necessary.
 *  @param new_buffer The new state.
 *  @param yyscanner The scanner object.
 */
void pcap_push_buffer_state (YY_BUFFER_STATE new_buffer , yyscan_t yyscanner)
{
    // 获取扫描器对象
    struct yyguts_t * yyg = (struct yyguts_t*)yyscanner;
    # 如果新缓冲区为空，则直接返回
    if (new_buffer == NULL)
        return;

    # 确保缓冲区栈的大小足够
    pcap_ensure_buffer_stack(yyscanner);

    # 以下代码块是从 pcap__switch_to_buffer 复制而来
    if ( YY_CURRENT_BUFFER )
        {
        # 清空旧缓冲区的信息
        *yyg->yy_c_buf_p = yyg->yy_hold_char;
        YY_CURRENT_BUFFER_LVALUE->yy_buf_pos = yyg->yy_c_buf_p;
        YY_CURRENT_BUFFER_LVALUE->yy_n_chars = yyg->yy_n_chars;
        }

    # 只有栈顶存在时才进行推入，否则替换栈顶
    if (YY_CURRENT_BUFFER)
        yyg->yy_buffer_stack_top++;
    YY_CURRENT_BUFFER_LVALUE = new_buffer;

    # 从 pcap__switch_to_buffer 复制而来
    pcap__load_buffer_state(yyscanner );
    yyg->yy_did_buffer_switch_on_eof = 1;
// 如果栈顶有元素，则移除并删除栈顶元素，下一个元素成为新的栈顶
// 参数 yyscanner 为扫描器对象
void pcap_pop_buffer_state (yyscan_t yyscanner)
{
    // 将 yyscanner 转换为 yyguts_t 结构体指针
    struct yyguts_t * yyg = (struct yyguts_t*)yyscanner;
    // 如果当前没有缓冲区，则返回
    if (!YY_CURRENT_BUFFER)
        return;
    // 删除当前缓冲区
    pcap__delete_buffer(YY_CURRENT_BUFFER ,yyscanner);
    // 将当前缓冲区置为 NULL
    YY_CURRENT_BUFFER_LVALUE = NULL;
    // 如果栈顶元素大于 0，则将栈顶指针减一
    if (yyg->yy_buffer_stack_top > 0)
        --yyg->yy_buffer_stack_top;
    // 如果当前有缓冲区
    if (YY_CURRENT_BUFFER) {
        // 加载缓冲区状态
        pcap__load_buffer_state(yyscanner );
        // 设置在 EOF 时进行缓冲区切换
        yyg->yy_did_buffer_switch_on_eof = 1;
    }
}

// 如果栈不存在，则分配栈空间
// 确保至少有一个推入的空间
static void pcap_ensure_buffer_stack (yyscan_t yyscanner)
{
    // 要分配的元素数量
    yy_size_t num_to_alloc;
    // 将 yyscanner 转换为 yyguts_t 结构体指针
    struct yyguts_t * yyg = (struct yyguts_t*)yyscanner;
    // 如果栈不存在
    if (!yyg->yy_buffer_stack) {
        // 第一次分配只为 2 个元素，因为我们不知道这个扫描器是否需要栈。我们使用 2 而不是 1，以避免在下一次调用时立即重新分配。
        num_to_alloc = 1; // 经过所有这些讨论，最终还是设置为 1...
        // 分配内存空间
        yyg->yy_buffer_stack = (struct yy_buffer_state**)pcap_alloc
                                (num_to_alloc * sizeof(struct yy_buffer_state*)
                                , yyscanner);
        // 如果分配失败，则抛出致命错误
        if ( ! yyg->yy_buffer_stack )
            YY_FATAL_ERROR( "out of dynamic memory in pcap_ensure_buffer_stack()" );
        // 将分配的内存空间清零
        memset(yyg->yy_buffer_stack, 0, num_to_alloc * sizeof(struct yy_buffer_state*));
        // 设置栈的最大容量为分配的数量
        yyg->yy_buffer_stack_max = num_to_alloc;
        // 设置栈顶指针为 0
        yyg->yy_buffer_stack_top = 0;
        return;
    }
}
    # 如果当前缓冲区栈的位置已经达到最大值减一，则需要增加缓冲区大小以准备可能的推送操作
    if (yyg->yy_buffer_stack_top >= (yyg->yy_buffer_stack_max) - 1){

        # 增加缓冲区的大小，准备可能的推送操作
        yy_size_t grow_size = 8 /* arbitrary grow size */;

        # 计算需要分配的缓冲区大小
        num_to_alloc = yyg->yy_buffer_stack_max + grow_size;
        # 重新分配缓冲区栈的内存空间
        yyg->yy_buffer_stack = (struct yy_buffer_state**)pcap_realloc
                                (yyg->yy_buffer_stack,
                                num_to_alloc * sizeof(struct yy_buffer_state*)
                                , yyscanner);
        # 如果分配失败，则抛出致命错误
        if ( ! yyg->yy_buffer_stack )
            YY_FATAL_ERROR( "out of dynamic memory in pcap_ensure_buffer_stack()" );

        # 将新分配的空间清零
        memset(yyg->yy_buffer_stack + yyg->yy_buffer_stack_max, 0, grow_size * sizeof(struct yy_buffer_state*));
        # 更新缓冲区栈的最大值
        yyg->yy_buffer_stack_max = num_to_alloc;
    }
}
/** Setup the input buffer state to scan directly from a user-specified character buffer.
 * @param base the character buffer
 * @param size the size in bytes of the character buffer
 * @param yyscanner The scanner object.
 * @return the newly allocated buffer state object. 
 */
YY_BUFFER_STATE pcap__scan_buffer  (char * base, yy_size_t  size , yyscan_t yyscanner)
{
    YY_BUFFER_STATE b;
    
    if ( size < 2 ||  // 如果缓冲区大小小于2
         base[size-2] != YY_END_OF_BUFFER_CHAR ||  // 或者倒数第二个字符不是结束符
         base[size-1] != YY_END_OF_BUFFER_CHAR )   // 或者最后一个字符不是结束符
        /* They forgot to leave room for the EOB's. */
        return 0;  // 返回空指针

    b = (YY_BUFFER_STATE) pcap_alloc(sizeof( struct yy_buffer_state ) ,yyscanner );  // 分配内存给缓冲区状态对象
    if ( ! b )  // 如果分配失败
        YY_FATAL_ERROR( "out of dynamic memory in pcap__scan_buffer()" );  // 报错

    b->yy_buf_size = size - 2;    /* "- 2" to take care of EOB's */  // 设置缓冲区大小，减去2是为了处理结束符
    b->yy_buf_pos = b->yy_ch_buf = base;  // 设置缓冲区位置和字符缓冲区
    b->yy_is_our_buffer = 0;  // 标记不是我们自己的缓冲区
    b->yy_input_file = 0;  // 输入文件为0
    b->yy_n_chars = b->yy_buf_size;  // 字符数等于缓冲区大小
    b->yy_is_interactive = 0;  // 标记不是交互式输入
    b->yy_at_bol = 1;  // 标记在行首
    b->yy_fill_buffer = 0;  // 填充缓冲区为0
    b->yy_buffer_status = YY_BUFFER_NEW;  // 缓冲区状态为新建

    pcap__switch_to_buffer(b ,yyscanner );  // 切换到新的缓冲区

    return b;  // 返回缓冲区状态对象
}

/** Setup the input buffer state to scan a string. The next call to pcap_lex() will
 * scan from a @e copy of @a str.
 * @param yystr a NUL-terminated string to scan
 * @param yyscanner The scanner object.
 * @return the newly allocated buffer state object.
 * @note If you want to scan bytes that may contain NUL values, then use
 *       pcap__scan_bytes() instead.
 */
YY_BUFFER_STATE pcap__scan_string (yyconst char * yystr , yyscan_t yyscanner)
{
    
    return pcap__scan_bytes(yystr,strlen(yystr) ,yyscanner);  // 返回从字符串扫描的缓冲区状态对象
}
# 设置输入缓冲区状态，以扫描给定的字节。下一次调用 pcap_lex() 将从 @a bytes 的副本中扫描。
# @param yybytes 要扫描的字节缓冲区
# @param _yybytes_len @a bytes 指向的缓冲区中的字节数。
# @param yyscanner 扫描器对象。
# @return 新分配的缓冲区状态对象。
YY_BUFFER_STATE pcap__scan_bytes  (yyconst char * yybytes, yy_size_t  _yybytes_len , yyscan_t yyscanner)
{
    YY_BUFFER_STATE b;
    char *buf;
    yy_size_t n;
    yy_size_t i;
    
    /* 为完整缓冲区获取内存，包括尾随 EOB 的空间。 */
    n = _yybytes_len + 2;
    buf = (char *) pcap_alloc(n ,yyscanner );
    if ( ! buf )
        YY_FATAL_ERROR( "out of dynamic memory in pcap__scan_bytes()" );

    for ( i = 0; i < _yybytes_len; ++i )
        buf[i] = yybytes[i];

    buf[_yybytes_len] = buf[_yybytes_len+1] = YY_END_OF_BUFFER_CHAR;

    b = pcap__scan_buffer(buf,n ,yyscanner);
    if ( ! b )
        YY_FATAL_ERROR( "bad buffer in pcap__scan_bytes()" );

    /* 可以扩展等此缓冲区，完成后应该丢弃它。 */
    b->yy_is_our_buffer = 1;

    return b;
}

/* User-visible API */
/* pcap_lex_init是特殊的，因为它创建了扫描器本身，所以它是唯一一个不将扫描器作为最后一个参数的可重入函数。这就是为什么我们明确处理声明，而不是使用我们的宏。 */

int pcap_lex_init(yyscan_t* ptr_yy_globals)

{
    // 检查传入的指针是否为空
    if (ptr_yy_globals == NULL){
        // 设置错误码并返回
        errno = EINVAL;
        return 1;
    }

    // 分配内存给扫描器
    *ptr_yy_globals = (yyscan_t) pcap_alloc ( sizeof( struct yyguts_t ), NULL );

    // 检查内存分配是否成功
    if (*ptr_yy_globals == NULL){
        // 设置错误码并返回
        errno = ENOMEM;
        return 1;
    }

    /* 通过设置为0xAA，我们暴露了yy_init_globals中的错误。发布时请设置为0x00。 */
    memset(*ptr_yy_globals,0x00,sizeof(struct yyguts_t));

    // 初始化全局变量
    return yy_init_globals ( *ptr_yy_globals );
}

/* pcap_lex_init_extra具有与pcap_lex_init相同的功能，但遵循将扫描器作为最后一个参数的约定。但需要注意的是，这是扫描器的*指针*，因为它将由此调用分配（也是这个函数必须处理自己的声明的原因）。第一个参数中的用户定义值将在yyextra字段中对pcap_alloc可用。 */

int pcap_lex_init_extra(YY_EXTRA_TYPE yy_user_defined,yyscan_t* ptr_yy_globals )

{
    struct yyguts_t dummy_yyguts;

    // 设置额外的用户定义值
    pcap_set_extra (yy_user_defined, &dummy_yyguts);

    // 检查传入的指针是否为空
    if (ptr_yy_globals == NULL){
        // 设置错误码并返回
        errno = EINVAL;
        return 1;
    }
    
    // 分配内存给扫描器
    *ptr_yy_globals = (yyscan_t) pcap_alloc ( sizeof( struct yyguts_t ), &dummy_yyguts );
    
    // 检查内存分配是否成功
    if (*ptr_yy_globals == NULL){
        // 设置错误码并返回
        errno = ENOMEM;
        return 1;
    }
    
    /* 通过设置为0xAA，我们暴露了yy_init_globals中的错误。发布时请设置为0x00。 */
    memset(*ptr_yy_globals,0x00,sizeof(struct yyguts_t));
    
    // 设置额外的用户定义值
    pcap_set_extra (yy_user_defined, *ptr_yy_globals);
    
    // 初始化全局变量
    return yy_init_globals ( *ptr_yy_globals );
}

static int yy_init_globals (yyscan_t yyscanner)
{
    // 将扫描器转换为yyguts_t结构体指针
    struct yyguts_t * yyg = (struct yyguts_t*)yyscanner;
    # 初始化与非可重入扫描器相同。
    # 这个函数是从pcap_lex_destroy()调用的，所以不要在这里分配内存。
    
    # 初始化扫描器的缓冲区栈
    yyg->yy_buffer_stack = 0;
    # 扫描器的缓冲区栈顶指针
    yyg->yy_buffer_stack_top = 0;
    # 扫描器的缓冲区栈最大值
    yyg->yy_buffer_stack_max = 0;
    # 扫描器的当前缓冲区指针
    yyg->yy_c_buf_p = (char *) 0;
    # 扫描器的初始化标志
    yyg->yy_init = 0;
    # 扫描器的起始位置
    yyg->yy_start = 0;
    
    # 扫描器的起始位置栈指针
    yyg->yy_start_stack_ptr = 0;
    # 扫描器的起始位置栈深度
    yyg->yy_start_stack_depth = 0;
    # 扫描器的起始位置栈
    yyg->yy_start_stack =  NULL;
/* Defined in main.c */
#ifdef YY_STDINIT
    // 如果定义了 YY_STDINIT，则将 yyin 和 yyout 分别指向标准输入和标准输出
    yyin = stdin;
    yyout = stdout;
#else
    // 如果未定义 YY_STDINIT，则将 yyin 和 yyout 分别指向空指针
    yyin = (FILE *) 0;
    yyout = (FILE *) 0;
#endif

    /* For future reference: Set errno on error, since we are called by
     * pcap_lex_init()
     */
    // 为将来的参考：在出错时设置 errno，因为我们是由 pcap_lex_init() 调用的
    return 0;
}

/* pcap_lex_destroy is for both reentrant and non-reentrant scanners. */
int pcap_lex_destroy  (yyscan_t yyscanner)
{
    struct yyguts_t * yyg = (struct yyguts_t*)yyscanner;

    /* Pop the buffer stack, destroying each element. */
    // 弹出缓冲区栈，销毁每个元素
    while(YY_CURRENT_BUFFER){
        pcap__delete_buffer(YY_CURRENT_BUFFER ,yyscanner );
        YY_CURRENT_BUFFER_LVALUE = NULL;
        pcap_pop_buffer_state(yyscanner);
    }

    /* Destroy the stack itself. */
    // 销毁栈本身
    pcap_free(yyg->yy_buffer_stack ,yyscanner);
    yyg->yy_buffer_stack = NULL;

    /* Destroy the start condition stack. */
    // 销毁起始条件栈
    pcap_free(yyg->yy_start_stack ,yyscanner );
    yyg->yy_start_stack = NULL;

    /* Reset the globals. This is important in a non-reentrant scanner so the next time
     * pcap_lex() is called, initialization will occur. */
    // 重置全局变量。这对于非可重入的扫描器非常重要，这样下一次调用 pcap_lex() 时会发生初始化
    yy_init_globals( yyscanner);

    /* Destroy the main struct (reentrant only). */
    // 销毁主结构体（仅适用于可重入的扫描器）
    pcap_free ( yyscanner , yyscanner );
    yyscanner = NULL;
    return 0;
}

/*
 * 将字符串转换为 32 位无符号整数。类似于 atoi()，但检查前导的 0x 或 0，并使用十六进制或八进制而不是十进制。
 *
 * 成功时，将 yylval->h 设置为该值并返回 NUM。
 * 失败时，设置 BPF 错误字符串并返回 LEX_ERROR，以强制停止解析。
 */
static int
stou(char *yytext_arg, YYSTYPE *yylval_arg, compiler_state_t *yyextra_arg)
{
    // 初始化变量 n 和 digit
    bpf_u_int32 n = 0;
    unsigned int digit;
    char *s = yytext_arg;

    /*
     * yytext_arg 被保证要么是一串十进制数字，要么是 0[xX] 后跟一串十六进制数字。
     */
    } else {
        /*
         * 十进制。
         */
        while ((digit = *s++) != '\0') {
            digit = digit - '0';
            // 定义 CUTOFF_DEC 和 CUTLIM_DEC
#define CUTOFF_DEC    (0xFFFFFFFFU / 10U)
#define CUTLIM_DEC    (0xFFFFFFFFU % 10U)
            // 检查是否溢出
            if (n > CUTOFF_DEC ||
                (n == CUTOFF_DEC && digit > CUTLIM_DEC)) {
                // 设置 BPF 错误信息
                bpf_set_error(yyextra_arg,
                    "number %s overflows 32 bits",
                    yytext_arg);
                return LEX_ERROR;
            }
            // 更新 n 的值
            n = (n * 10) + digit;
        }
    }

    // 将结果赋值给 yylval->h，并返回 NUM
    yylval_arg->h = n;
    return NUM;
}
